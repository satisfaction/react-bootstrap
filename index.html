TextInput (Validation)

function validateTextInput(state) {
  if (!/^^(?:\(\d+\))?\s?[\d\-]+$/.test(state.value)) {
    return 'Invalid phone number';
  }
}

<ReactUI.TextInput placeholder="Phone Number" validate={validateTextInput} />
